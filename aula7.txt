DATA CONTROL LANGUAGE - SQL

CREATE USER Nome - cria usuario
DROP USER Nome - deleta usuário
GRANT - habilita acesso
REVOKE - remove acesso

sintaxe GRANT  
	GRANT Ação ON Estrutura TO Usuário

sintaxe REVOKE 
	REVOKE Ação ON Estrutura FROM Usuário
AÇÕES
	all, select, insert, update, delete.
ESTRUTURAS
	Tables, View, sequences.
